Configuration file grammar:
//lowercase for literal, capital for production rule.
START -> DECK PROBABILITY .

DECK -> DECK_START open_brace DECK_LIST close_brace.
DECK_START -> deck_list: .
DECK_LIST -> CARD MORE_CARDS .
MORE_CARDS -> CARD MORE_CARDS | .
CARD -> NAME ; .
NAME -> name .

PROBABILITY -> PROBABILITY_START open_brace SCENARIO_LIST close_brace .
PROBABILITY_START -> scenarios: .

SCENARIO_LIST -> SCENARIO MORE_SCENARIOS .
MORE_SCENARIOS -> SCENARIO MORE_SCENARIOS | .

SCENARIO -> SCENARIO_NAME open_brace TREE DISPLAY close_brace .
SCENARIO_NAME -> NAME equals .

TREE -> TREE_START open_brace EXPR close_brace .
TREE_START -> scenario equals .

EXPR -> UNARY_EXPR BINARY_EXPR .
PRIMARY_EXPR -> open_bracket NAME close_bracket 
               |less_than NAME greater_than 
               |open_parenthesis EXPR close_parenthesis .

UNARY_EXPR -> UNARY_OPERATOR UNARY_EXPR | PRIMARY_EXPR .
UNARY_OPERATOR -> not .

BINARY_EXPR -> BINARY_OPERATOR UNARY_EXPR BINARY_EXPR | .
BINARY_OPERATOR -> and | or | xor .

DISPLAY -> DISPLAY_START DISPLAY_VALUE ; | .
DISPLAY_START -> display equals .
DISPLAY_VALUE -> true | false .
