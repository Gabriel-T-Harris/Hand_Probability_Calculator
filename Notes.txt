START -> DECK PROBABILITY
DECK -> "Deck list:{" CARD ENTRY //deck list
ENTRY -> CARD ENTRY | "}" //any number of cards
CARD -> NAME ";" //card

PROBABILITY -> "Scenarios:{" SCENARIO MORE //stuff to be calculated
MORE -> SCENARIO MORE | "}" //any number scenarios
SCENARIO -> SCENARIO_NAME TREE DISPLAY "}"//scenario
SCENARIO_NAME -> NAME "= {" //name of scenario
TREE -> "Scenario = {" EXPRESSION "}"//probability being calculated
DISPLAY -> "Display = " ("true" | "false") ";" | NULL //optional

EXPRESSION -> NEGATE "(" EXPRESSION ")" | PART OPERATOR
PART -> NEGATE ("[" NAME "]" | "<" NAME ">") //brackets for card and angular braces for scenario
NEGATE -> "NOT" | NULL //optional
OPERATOR -> ("OR" | "AND" | "XOR") (PART | EXPRESSION) | NULL
